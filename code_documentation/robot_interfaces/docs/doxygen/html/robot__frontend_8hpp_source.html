152  TimeIndex append_desired_action(const Action &desired_action)
153  {
154  // check error state. do not allow appending actions if there is an
155  // error
156  if (robot_data_->status->length() > 0)
157  {
158  const Status status = robot_data_->status->newest_element();
159  switch (status.error_status)
160  {
161  case Status::ErrorStatus::NO_ERROR:
162  break;
163  case Status::ErrorStatus::DRIVER_ERROR:
164  throw std::runtime_error("Driver Error: " +
165  status.error_message);
166  case Status::ErrorStatus::BACKEND_ERROR:
167  throw std::runtime_error("Backend Error: " +
168  status.error_message);
169  default:
170  throw std::runtime_error("Unknown Error: " +
171  status.error_message);
172  }
173  }
174 
175  // since the timeseries has a finite memory, we need to make sure that
176  // by appending new actions we do not forget about actions which have
177  // not been applied yet
178  if (robot_data_->desired_action->length() ==
179  robot_data_->desired_action->max_length() &&
180  robot_data_->desired_action->oldest_timeindex() == // FIXME >=
181  get_current_timeindex())
182  {
183  std::cout
184  << "you have been appending actions too fast, waiting for "
185  "RobotBackend to catch up with executing actions."
186  << std::endl;
187  wait_until_timeindex(
188  robot_data_->desired_action->oldest_timeindex() + 1);
189  }
190 
191  robot_data_->desired_action->append(desired_action);
192  return robot_data_->desired_action->newest_timeindex();
193  }
