64  MonitoredRobotDriver(RobotDriverPtr robot_driver,
65  const double max_action_duration_s,
66  const double max_inter_action_duration_s)
67  : robot_driver_(robot_driver),
68  max_action_duration_s_(max_action_duration_s),
69  max_inter_action_duration_s_(max_inter_action_duration_s),
70  is_shutdown_(false),
71  action_start_logger_(1000),
72  action_end_logger_(1000)
73  {
74  thread_ = std::make_shared<real_time_tools::RealTimeThread>();
75 
76  // if both timeouts are infinite, there is no need to start the loop at
77  // all
78  if (std::isfinite(max_action_duration_s_) &&
79  std::isfinite(max_inter_action_duration_s_))
80  {
81  thread_->create_realtime_thread(&MonitoredRobotDriver::loop, this);
82  }
83  else
84  {
85  std::cerr
86  << "WARNING: MonitoredRobotDriver was created with a "
87  "non-finite timeout. The monitoring loop is NOT executed. "
88  "If monitoring is not needed, consider using the driver "
89  "directly without the MonitoredRobotDriver-wrapper."
90  << std::endl;
91  }
92  }

113  virtual typename Driver::Action apply_action(
114  const typename Driver::Action &desired_action) final
115  {
116  if (is_shutdown_)
117  {
118  // FIXME I don't think it makes sense to return the desired action

190  void loop()
191  {
192  // wait for the first data
193  while (!is_shutdown_ &&
194  !action_start_logger_.wait_for_timeindex(0, 0.1))
195  {
196  }
197 
198  // loop until shutdown and monitor action timing
199  for (size_t t = 0; !is_shutdown_; t++)
200  {
201  bool action_has_ended_on_time =
202  action_end_logger_.wait_for_timeindex(t,
203  max_action_duration_s_);
204  if (!action_has_ended_on_time)
205  {
206  error_message_.set(
207  "Action did not end on time, shutting down.");
208  shutdown();
209  return;
210  }
211  bool action_has_started_on_time =
212  action_start_logger_.wait_for_timeindex(
213  t + 1, max_inter_action_duration_s_);
214  if (!action_has_started_on_time)
215  {
216  error_message_.set(
217  "Action did not start on time, shutting down.");
218  shutdown();
219  return;
220  }
221  }
222  }

227  return nullptr;
228  }
