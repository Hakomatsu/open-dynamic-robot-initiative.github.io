232  void loop()
233  {
234  const double start_time =
235  real_time_tools::Timer::get_current_time_sec();
236 
237  // wait until first desired_action was received
238  // ----------------------------
239  while (!has_shutdown_request() &&
240  !robot_data_->desired_action->wait_for_timeindex(0, 0.1))
241  {
242  const double now = real_time_tools::Timer::get_current_time_sec();
243  if (now - start_time > first_action_timeout_)
244  {
245  Status status;
246  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
247  "First action was not provided in time");
248 
249  robot_data_->status->append(status);
250 
251  std::cerr << "Error: " << status.error_message
252  << "\nRobot is shut down." << std::endl;
253 
254  request_shutdown();
255  break;
256  }
257  }
258 
259  for (long int t = 0; !has_shutdown_request(); t++)
260  {
261  // TODO: figure out latency stuff!!
262 
263  Status status;
264 
265  if (max_number_of_actions_ > 0 && t >= max_number_of_actions_)
266  {
267  // TODO this is not really an error
268  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
269  "Maximum number of actions reached.");
270  }
271 
272  timer_.start();
273 
274  // get latest observation from robot and append it to robot_data_
275  Observation observation = robot_driver_->get_latest_observation();
276  timer_.checkpoint("get observation");
277 
278  robot_data_->observation->append(observation);
279  // TODO: for some reason this sometimes takes more than 2 ms
280  // i think this may be due to a non-realtime thread blocking the
281  // timeseries. this is in fact an issue, we might have to
282  // duplicate all the timeseries and have a realtime thread
283  // writing back and forth
284  timer_.checkpoint("append observation");
285 
286  // If real time mode is enabled the next action needs to be provided
287  // in time. If this is not the case, optionally repeat the previous
288  // action or raise an error.
289  if (real_time_mode_ &&
290  robot_data_->desired_action->newest_timeindex() < t)
291  {
292  uint32_t action_repetitions =
293  robot_data_->status->newest_element().action_repetitions;
294 
295  if (action_repetitions < max_action_repetitions_)
296  {
297  robot_data_->desired_action->append(
298  robot_data_->desired_action->newest_element());
299  status.action_repetitions = action_repetitions + 1;
300  }
301  else
302  {
303  // No action provided and number of allowed repetitions
304  // of the previous action is exceeded --> Error
305  status.set_error(Status::ErrorStatus::BACKEND_ERROR,
306  "Next action was not provided in time");
307  }
308  }
309 
310  std::string driver_error_msg = robot_driver_->get_error();
311  if (!driver_error_msg.empty())
312  {
313  status.set_error(Status::ErrorStatus::DRIVER_ERROR,
314  driver_error_msg);
315  }
316 
317  robot_data_->status->append(status);
318 
319  // if there is an error, shut robot down and stop loop
320  if (status.error_status != Status::ErrorStatus::NO_ERROR)
321  {
322  std::cerr << "Error: " << status.error_message
323  << "\nRobot is shut down." << std::endl;
324  break;
325  }
326  timer_.checkpoint("status");
327 
328  // early exit if destructor has been called
329  while (!has_shutdown_request() &&
330  !robot_data_->desired_action->wait_for_timeindex(t, 0.1))
331  {
332  }
333  if (has_shutdown_request())
334  {
335  break;
336  }
337 
338  Action desired_action = (*robot_data_->desired_action)[t];
339  timer_.checkpoint("get action");
340 
341  Action applied_action = robot_driver_->apply_action(desired_action);
342  timer_.checkpoint("apply action");
343 
344  robot_data_->applied_action->append(applied_action);
345  timer_.checkpoint("append applied action");
346 
347  if (t % 5000 == 0 && t > 0)
348  {
349  timer_.print_statistics();
350  }
351  }
352 
353  robot_driver_->shutdown();
354  loop_is_running_ = false;
355  }
