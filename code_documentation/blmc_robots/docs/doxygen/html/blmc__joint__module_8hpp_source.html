640  HomingReturnCode execute_homing(
641  double search_distance_limit_rad,
642  Vector home_offset_rad,
643  Vector profile_step_size_rad = Vector::Constant(0.001))
644  {
645  // Initialise homing for all joints
646  for (size_t i = 0; i < COUNT; i++)
647  {
648  modules_[i]->init_homing((int)i,
649  search_distance_limit_rad,
650  home_offset_rad[i],
651  profile_step_size_rad[i]);
652  }
653 
654  // run homing for all joints until all of them are done
655  real_time_tools::Spinner spinner;
656  spinner.set_period(0.001); // TODO magic number
657  HomingReturnCode homing_status;
658  do
659  {
660  bool all_succeeded = true;
661  homing_status = HomingReturnCode::RUNNING;
662 
663  for (size_t i = 0; i < COUNT; i++)
664  {
665  HomingReturnCode joint_result = modules_[i]->update_homing();
666 
667  all_succeeded &= (joint_result == HomingReturnCode::SUCCEEDED);
668 
669  if (joint_result == HomingReturnCode::NOT_INITIALIZED ||
670  joint_result == HomingReturnCode::FAILED)
671  {
672  homing_status = joint_result;
673  // abort homing
674  break;
675  }
676  }
677  if (homing_status == HomingReturnCode::RUNNING)
678  {
679  for (unsigned i = 0; i < COUNT; ++i)
680  {
681  modules_[i]->send_torque();
682  }
683  }
684 
685  if (all_succeeded)
686  {
687  homing_status = HomingReturnCode::SUCCEEDED;
688  }
689 
690  spinner.spin();
691  } while (homing_status == HomingReturnCode::RUNNING);
692 
693  return homing_status;

715  GoToReturnCode go_to(Vector angle_to_reach_rad,
716  double average_speed_rad_per_sec = 1.0)
717  {
718  // Compute a min jerk trajectory
719  Vector initial_joint_positions = get_measured_angles();
720  double final_time = (angle_to_reach_rad - initial_joint_positions)
721  .array()
722  .abs()
723  .maxCoeff() /
724  average_speed_rad_per_sec;
725 
726  std::array<TimePolynome<5>, COUNT> min_jerk_trajs;
727  for (unsigned i = 0; i < COUNT; i++)
728  {
729  min_jerk_trajs[i].set_parameters(final_time,
730  initial_joint_positions[i],
731  0.0 /*initial speed*/,
732  angle_to_reach_rad[i]);
733  }
734 
735  // run got_to for all joints
736  real_time_tools::Spinner spinner;
737  double sampling_period = 0.001; // TODO magic number
738  spinner.set_period(sampling_period);
739  GoToReturnCode go_to_status;
740  double current_time = 0.0;
741  do
742  {
743  // TODO: add a security if error gets too big
744  for (unsigned i = 0; i < COUNT; i++)
745  {
746  double desired_pose = min_jerk_trajs[i].compute(current_time);
747  double desired_torque =
748  modules_[i]->execute_position_controller(desired_pose);
749  modules_[i]->set_torque(desired_torque);
750  }
751  for (unsigned i = 0; i < COUNT; ++i)
752  {
753  modules_[i]->send_torque();
754  }
755  go_to_status = GoToReturnCode::RUNNING;
756 
757  current_time += sampling_period;
758  spinner.spin();
759 
760  } while (current_time < (final_time + sampling_period));
761 
762  // Stop all motors (0 torques) after the destination achieved
763  for (unsigned i = 0; i < COUNT; i++)
764  {
765  modules_[i]->set_torque(0.0);
766  }
767  for (unsigned i = 0; i < COUNT; ++i)
768  {
769  modules_[i]->send_torque();
770  }
771 
772  Vector final_pos = get_measured_angles();
773  if ((angle_to_reach_rad - final_pos).isMuchSmallerThan(1.0, 1e-3))
774  {
775  go_to_status = GoToReturnCode::SUCCEEDED;
776  }
777  else
778  {
779  go_to_status = GoToReturnCode::FAILED;
780  }
781  return go_to_status;
